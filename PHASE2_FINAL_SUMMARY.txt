================================================================================
                          CLI USAGE GUIDE
================================================================================

INSTALLATION:
  $ source venv/bin/activate
  $ python3 -m pip install -r requirements.txt

BASIC USAGE:
  $ python3 analyze_vulnerabilities_with_ollama.py scan.jsonl

ENABLE CACHING:
  $ python3 analyze_vulnerabilities_with_ollama.py scan.jsonl --use-cache

VIEW CACHE STATS:
  $ python3 analyze_vulnerabilities_with_ollama.py --cache-stats

CLEAR CACHE:
  $ python3 analyze_vulnerabilities_with_ollama.py scan.jsonl --clear-cache

FILTERING:
  $ python3 analyze_vulnerabilities_with_ollama.py scan.jsonl \
    --severity critical,high

MULTI-FORMAT REPORTS:
  $ python3 analyze_vulnerabilities_with_ollama.py scan.jsonl \
    --format html -o report.html

COMBINED FEATURES:
  $ python3 analyze_vulnerabilities_with_ollama.py scan.jsonl \
    --severity critical,high \
    --use-cache \
    --format html -o report.html

COMPREHENSIVE ANALYSIS:
  $ python3 analyze_vulnerabilities_with_ollama.py scan.jsonl \
    --all-analysis \
    --use-cache \
    --format html \
    -o comprehensive.html

SHOW HELP:
  $ python3 analyze_vulnerabilities_with_ollama.py --help

SHOW EXAMPLES:
  $ python3 analyze_vulnerabilities_with_ollama.py --examples
